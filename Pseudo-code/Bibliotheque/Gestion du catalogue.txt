FUNCTION ajouter_livre(livre)
	AJOUTER livre ->  livres
Fin FUNCTION

FUNCTION modifier_livre(isbn, nouvelles_infos)
	POUR CHANQUE livre DANS livres
		SI livre['isbn'] == isbn ALORS
			METTRE À JOUR livre AVEC nouvelles_infos
			RETORNER VRAI
		FIN SI
	FIN POUR
	RETORNER FRAUX
Fin FUNCTION

FUNCTION supprimer_livre(isbn)
	POUR CHANQUE livre DANS livres
		SI livre['isbn'] == isbn ALORS
			SUPPRIMER livre DE livres
			RETORNER VRAI
		FIN SI
	FIN POUR
	RETORNER FAUX
FIN FUNCTION
